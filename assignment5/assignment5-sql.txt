SELECT orders.order_id, SUM(line_items.quantity * products.price) as total_price FROM orders    
JOIN line_items ON orders.order_id=line_items.order_id  
JOIN products ON products.product_id=line_items.product_id 
GROUP BY orders.order_id 
ORDER BY orders.order_id 
LIMIT 5 ;

#5a-2#
SELECT customers.customer_name, AVG(t.total_price) as average_order_price FROM customers 
JOIN (SELECT orders.customer_id, orders.order_id, SUM(line_items.quantity * products.price) as total_price FROM orders    
JOIN line_items ON orders.order_id=line_items.order_id  
JOIN products ON products.product_id=line_items.product_id 
GROUP BY orders.order_id ) as t ON t.customer_id=customers.customer_id 
GROUP BY customers.customer_id, customers.customer_name 
ORDER BY customers.customer_name;
